  'January',
  'February',
  'March',
  'April',
  'May',
  'June',
  'July',

  /**
   *
   *
   * @memberof CartForm
   */
  getMonthOptions = () => {
    const monthOptions = {};
    [...Array(12).keys()].forEach(v => {
      monthOptions[v] = months[v];
    });
    return monthOptions;
  };

  /**
   *
   *
   * @memberof CartForm
   */
  getYearOptions = () => {
    const yearOptions = {};
    [18, 19, 20, 21, 22, 23, 24].forEach(v => {
      yearOptions[`20${v}`] = v;
    });
    return yearOptions;
  };

  /**
   *

   * @memberof CartForm

  _toggleCVVModal = e => {
    e.preventDefault();
    this.setState({ show_cvv_modal: !this.state.show_cvv_modal });
